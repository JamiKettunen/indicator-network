 
find_package(QmlPlugins)

  "${CMAKE_SOURCE_DIR}/src/connectivity-api/connectivity-qt"
)

)

  ${CONNECTIVITY_QT_LIB_TARGET}
  Qt5::Qml
)

  Ubuntu.Connectivity
  1.0
  Ubuntu/Connectivity
  TARGET_PREFIX Connectivity
  TARGETS connectivity-qml
)
